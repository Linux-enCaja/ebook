\section{Situación Colombiana}


\begin{frame}
  \frametitle{Situación Colombiana \cite{LV98}} 
     \begin{itemize}
      \item Según el ministerio de educación en el país existen 297 programas relacionados con la electrónica. 
      \item Según ASESEL en el 2001 existían 154 empresas productoras de componentes y equipos de la cadena electrónica.
        \begin{itemize}
         \item 76\% usa tecnología de los 80 o anterior
         \item Las empresas no adquieren el conocimiento necesario para innovar; se limitan a compra de equipo.
        \end{itemize}
      \item Según el DNP, DDE, basados en información del DANE y la DIAN, las importaciones son 64 veces mayores que las exportaciones.
      \item \alert{Principal problema: tecnología de producción desactualizada.}

\subsection{Obstáculos para la Transferencia} 
\begin{frame}
  \frametitle{Obstáculos para la Transferencia \cite{MO90}}
    \begin{itemize}
     \item  Falta de facilidades en educación y en capacitación. 
     \item  La resistencia o el desconocimiento a la tecnología.
     \item  La transferencia es efectiva solo si la economía en la cual es introducida es capaz de utilizarla
     \item  \textcolor{DarkGreen}{Falta de personal disponible y dispuesto a absorber el \textit{know-how} asociado a la tecnología}.
     \item  Competencia de países asiáticos.
    \end{itemize}
\end{frame}

\section{Objetivos}
\begin{frame}
  \frametitle{Objetivos}

%    \setbeamercolor{block title}{bg=red,fg=blue}%bg=background, fg= foreground
   \setbeamercolor{block body}{bg=yellow!10}%bg=background, fg= foreground

   \only<1>{
     \begin{alertblock}{Objetivo Principal}
       Desarrollar y aplicar una metodología para la transferencia tecnológica y de conocimientos en el diseño de Sistemas Embebidos y de esta forma contribuir a dar solución al problema del atraso tecnológico en Colombia.
     \end{alertblock}
   }

   \only<2>{
     \begin{block}{Objetivos Específicos: Asimilación de conocimientos}
       \begin{itemize}
         \small
         \item \textcolor{DarkGreen}{Formulación/adaptación y aplicación de una metodología para la transferencia tecnológica y de conocimientos} en el área de diseño y fabricación de Sistemas Embebidos en Colombia que permita generar cambios globales a partir de interacciones locales.

         \item \textcolor{DarkGreen}{Formulación o adaptación de una metodología de diseño y producción para sistemas embebidos} aplicable en el entorno local y su respectiva aplicación y validación en un programa académico de un centro de formación superior consolidado y en una empresa de base tecnológica.
       \end{itemize}
     \end{block}
   }
   \only<3->{
     \begin{block}{Objetivos Específicos: Creación de habilidades}
       \begin{itemize}

         \item \textcolor{DarkGreen}{Identificación de las habilidades requeridas por los profesionales en ingeniería electrónica} para estar acorde con la tendencia de la industria electrónica mundial. \textcolor{DarkGreen}{Creación de un programa académico} que ayude al refuerzo de estas habilidades en las asignaturas relacionadas con el diseño digital.

         \item \textcolor{DarkGreen}{Generación de un recurso público} basado en el conocimiento necesario para diseñar implementar y producir sistemas digitales.

         \item \textcolor{DarkGreen}{Creación de herramientas de difusión} que permitan el acceso de este recurso a cualquier sector de la sociedad que esté interesado.
       \end{itemize}
     \end{block}
   }
