\subsection{Situación Colombiana}

\begin{frame}
  \frametitle{Situación Colombiana \cite{LV98}} 
    \begin{itemize}
      \item Según ASESEL\footnote{Asociacion de Entidades del Sector Electronico} en el 2001 existían 154 empresas productoras de componentes y equipos de la cadena electrónica.
        \begin{itemize}
         \item 76\% usa tecnología de los 80 o anterior
        \end{itemize}
      \item Las empresas no adquieren el conocimiento necesario para innovar; se limitan a compra de equipo.
      \item Principal problema: tecnología de producción desactualizada.
      \item Según el DNP\footnote{Departamento nacional de planeación}, DDE\footnote{Dirección de desarrollo empresarial}, basados en información del DANE\footnote{Departamento Administrativo Nacional de Estadistica} y la DIAN\footnote{Dirección de Impuestos y Aduanas Nacionales}, las importaciones son 64 veces mayores que las exportaciones.
      \item En el país existen 297 programas relacionados con la electrónica. 
    \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Causas del Atraso}
    \begin{itemize}
     \item Pobre enfoque académico hacia la industria.
     \item Baja calidad de los productos nacionales.
     \item Falta de políticas gubernamentales.
     \item Falta de cultura de investigación y reducida apropiación tecnológica.
     \item \alert{Limitado recurso humano con formación adecuada}.
    \end{itemize}
\end{frame}

\subsection{Recomendaciones a la Academia}
\begin{frame}
\frametitle{Recomendaciones para la academia}
\begin{itemize}
    \item Actualización curricular.
     \begin{itemize}
        \item \textcolor{Navy}{Mejorar las competencias y habilidades generales de los ingenieros.}
        \item \textcolor{DarkGreen}{Maestrías y doctorados nacionales conectados con el sector productivo.}
        \item Difusión de conocimientos entre todos los estratos de la población.
     \end{itemize}
    \item Alianza con la industria
     \begin{itemize}
       \item \textcolor{Navy}{Desarrollar habilidades y competencias que la empresa requiere.}
       \item Buscar fortalezas en áreas dominadas por las industrias locales.
     \end{itemize}
    \item Promover y Soportar la Transferencia Tecnológica.
     \begin{itemize}
       \item Investigación aplicada orientada a mejorar la productividad empresarial.
       \item \textcolor{DarkGreen}{Impulsar la actualización tecnológica desarrollando proyectos con una posible transferencia de tecnología.}
     \end{itemize}
    \item Búsqueda de financiación para investigación y desarrollo
\end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Transferencia Tecnológica}
 
  \begin{alertblock}{}
    \begin{itemize}
     \item Odedra \cite{Mo94}: \textbf{La transferencia tecnológica se considera exitosa cuando los receptores de la tecnología \textcolor{DarkGreen}{asimilan} estos conceptos para \textcolor{DarkGreen}{suplir sus necesidades locales} generando productos novedosos}.
     \item Jolly \cite{Jol77}: \textbf{El conocimiento es lo que queda al final de un proceso \textcolor{DarkGreen}{documentado y difundido} de forma apropiada}.
    \end{itemize}
  \end{alertblock}

\end{frame}

\begin{frame}
  \frametitle{Canales para la TT}
   \begin{center} \includegraphics[scale=.6]{../images/TT_canales.pdf}   \end{center}
\end{frame}


\begin{frame}
  \frametitle{Obstáculos para la Transferencia \cite{MO90}}
    \begin{itemize}
     \item  Falta de facilidades en educación y en capacitación. 
     \item  La resistencia o el desconocimiento a la tecnología.
     \item  La transferencia es efectiva solo si la economía en la cual es introducida es capaz de utilizarla
     \item  Falta de políticas claras.
     \item  Competencia de países asiáticos.
     \item  \textcolor{DarkGreen}{Falta de personal disponible y dispuesto a absorber el \textit{know-how} asociado a la tecnología}.
    \end{itemize}
\end{frame}
